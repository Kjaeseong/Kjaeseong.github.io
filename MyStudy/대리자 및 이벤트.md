
- void 타입의 함수만 보관할 수 있음
- 타입을 정의
- 객체가 아니다
```cs
//타입 정의
private delegate void SecondElapsedDelegate();

//SecondElapsedDelegate타입의 객체 OnSecondElapsed 생성
public SecondElapsedDelegate OnSecondElapsed;

//OnSecondElapsed에는 void 타입의 함수를 넣을 수 있음
//int 타입의 함수만
delegate int Foo();
OnSecondElapsed += Foo;

delegate void EventHandler(string message);
// void(string) 타입의 함수를 보관할 수 있는 대리자 이벤트핸들러 타입 정의

EventHandler eventHandler;
eventHandler += Foo; //오류
// Foo 타입은 voiid()
// 이벤트핸들러 대리자 타입이 담을 수 있는 함수는 void(string)


//float 타입 함수만

delegate float Boo(int, double);
```
